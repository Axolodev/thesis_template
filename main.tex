% APA 6th edition first/subsequent citation rule for author names:
% - 2 authors: always both names
% - 3-5 authors: first citation shows all; subsequent uses first author + et al.
% - 6+ authors: always first author + et al.
%
% Strategy: maxcitenames=5 on first citation (shows ≤5 authors, truncates 6+)
%           maxcitenames=2 on subsequent (shows ≤2 authors, truncates
% 3+ to et al.)
\makeatletter
\AtEveryCitekey{%
  \edef\blx@thiskey{\thefield{entrykey}}%
  \ifcsundef{blx@apa@seen@\blx@thiskey}
  {%  % First citation: mark as seen (only during rendering pass) and
    % show up to 5 authors
    \ifbool{citetracker}{\csgdef{blx@apa@seen@\blx@thiskey}{}}{}%
    \setcounter{maxnames}{5}%
  }%
  {%  % Subsequent citation: truncate 3+ authors to et al.
    \setcounter{maxnames}{2}%
  }%
}
\makeatother
